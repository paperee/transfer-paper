<script>
    interval=[]
    now={}
    event={}
    config={
        color:"#555",
        text:"你还活着",
    }
    title={
        space:[8,8],
        text:[[2,2],"《文字游戏》","#335469"],
        menu:[
            [[[5,3],"开始游戏","#114514"],[[false,()=>goto("lounge",[2,3])]]],
            [[[7,3],"继续游戏","#888"]]
        ]
    }
    tile={
        space:["　",""],
        floor:["地","纸制的地板","#D2DDDD"],
        wall:["墙","纸制的墙"],
        tree:["树","绿油油的树","#114514"],
        water:["池","池塘里的水","#89ABCD"],
        door:["门","纸制的门"],
        hole:["洞","一个很大的洞"],
        bed:["床","纸制的床"],
        table:["桌","纸制的桌子","#AA8E57"],
        chair:["椅","纸制的椅子","#8A7E57"],
        paper:["纸","一张普通的纸","#4F0"],
        dog:["狗","一条普通的狗","#456789"],
        grass:["草","#44FF00色的草","#4F0"],
        zhang:["张","你的朋友小张","#4F0"],
        maggie:["鸡","你养的麻鸡","#4F0"],
        fish:["鱼","你养的西瓜鱼","#F40"],
        you:["你","你是主角","#335469"]
    }
    world={
        lounge:{
            space:[8,8,"floor"],
            bound:[true,"wall"],
            door:[[[1,3],[[false,()=>goto("home",[2,3])]]]],
            dog:[
                [[3,4],[[false,()=>log("一条名叫23的公狗")],[true,(w)=>move(w,0,1400)]]],
                [[4,5],[[false,()=>log("一条名叫xw的母狗")],[true,(w)=>move(w,1,1200)]]],
                [[5,3],[[true,(w)=>move(w,2,1000)]]],
                [[6,6],[[true,(w)=>move(w,3,800)]]],
                [[7,2],[[true,(w)=>move(w,4,600)]]]
            ]
        },
        home:{
            space:[8,8,"floor"],
            bound:[true,"wall"],
            door:[[[1,3],[[false,()=>goto("lounge",[2,3])]]]],
            hole:[[[4,8],[[false,()=>goto("garden",[4,7])]]]],
            table:[[[6,4]],[[6,5]]],
            chair:[[[6,3]],[[6,6]]],
            paper:[[[4,3],[[true,(w)=>move(w,0,800)]]]],
            zhang:[[[2,7],[[true,(w)=>move(w,0,600)]]]],
            maggie:[[[3,5],[[true,(w)=>move(w,0,400)]]]],
            fish:[[[2,2]]]
        },
        garden:{
            space:[8,8,"grass"],
            bound:[true,"tree"],
            water:[[[2,7]],[[2,6]],[[2,5]],[[3,7]],[[3,6]]],
            hole:[[[4,8],[[false,()=>goto("home",[4,7])]]]],
        }
    }
    log=(text,color=undefined)=>{
        $("#instruction").text(text).css("color",color)
    }
    press=(word,name,text=undefined,color=undefined)=>{
        return word.text(text||tile[name][0]).attr("class",name).css("color",color||tile[name][2]||config.color)
    }
    lookup=(p)=>{
        return $(`[ln="${p[0]}"]`).filter(function () {
            return $(this).attr("col")==`${p[1]}`
        })
    }
    clear=()=>{
        $("game").html("")
        $("#room").text(now.scene)
        interval.forEach((value)=>clearInterval(value))
        interval=[]
        event={}
    }
    space=(p,you)=>{
        while (true) {
            if (!lookup(p).text()||p.toString()==you.toString()) {
                p[Math.random()<0.5?1:0]+=Math.random()<0.5?1:-1
            }
            else break
        }
        return p
    }
    key=(key)=>{
        if ($(".you").length) {
            oldP=[parseInt($(".you").attr("ln")),parseInt($(".you").attr("col"))]
            newP=oldP.slice()
            if (key==37) newP[1]-=1 // 左
            else if (key==38) newP[0]-=1 // 上
            else if (key==39) newP[1]+=1 // 右
            else if (key==40) newP[0]+=1 // 下
            else return
            newW=lookup(newP)
            str=newP.toString()
            tile_=tile[newW.attr("class")]
            log(tile_[1]||config.text,tile_[2]||config.color)
            if (newW.attr("class")==now.space) {
                press(lookup(oldP),now.space)
                press(newW,"you")
            }
            else if (Object.keys(event).includes(str)) {
                event[str].forEach((value)=>value())
            }
            return key
        }
        else {
            if (key==13||key==32) {}
        }
    }
    move=(word,index,speed)=>{
        interval.push(setInterval(()=>{
            oldP=[parseInt(word.attr("ln")),parseInt(word.attr("col"))]
            newP=oldP.slice()
            newP[Math.random()<0.5?1:0]+=Math.random()<0.5?1:-1
            newW=lookup(newP)
            str=newP.toString()
            name=word.attr("class")
            if (newW.attr("class")==now.space) {
                press(lookup(oldP),now.space)
                press(newW,name)
                word=lookup(newP)
                world[now.scene][name][index][0]=newP
                if (Object.keys(event).includes(oldP.toString())) {
                    event[newP]=event[oldP]
                    delete event[oldP]
                }
            }
        },speed))
    }
    fill=(space)=>{
        for (i=1;i<=space[1];i++) {
            for (e=1;e<=space[0];e++) {
                now.space=space[2]||"space"
                press($(`<e class="space" ln="${i}" col="${e}">`).appendTo($("game")),now.space)
            }
            $("<br>").appendTo($("game"))
        }
    }
    goto=(scene,p)=>{
        block=world[scene]
        now.scene=scene
        clear()
        Object.keys(block).forEach((value)=>{
            if (value=="space") fill(block.space)
            else if (value=="bound") {
                if (block["bound"][0]) {
                    press($(`[ln="1"],[ln="${block.space[0]}"],[col="1"],[col="${block.space[1]}"]`),block["bound"][1])
                }
            }
            else {
                block[value].forEach((value_)=>{
                    str=value_[0].toString()
                    if (str==p.toString()) value_[0]=space(value_[0],p)
                    word=press(lookup(value_[0]),value)
                    if (value_[1]) {
                        value_[1].forEach((value__)=>{
                            if (value__[0]) value__[1](word)
                            else if (Object.keys(event).includes(str)) event[str].push(value__[1])
                            else event[str]=[value__[1]]
                        })
                    }
                })
            }
        })
        press(lookup(p),"you")
    }
    home=()=>{
        Object.keys(title).forEach((value)=>{
            if (value=="space") fill(title.space)
            else if (value=="text") {
                list=(title.text[1]).split("")
                for (i=0;i<list.length;i++) {
                    press(lookup([title.text[0][0],title.text[0][1]+i]),"text",list[i],title.text[2]||config.color)
                }
            }
            else if (value=="menu") {
                title.menu.forEach((value_,index)=>{
                    list=value_[0][1].split("")
                    ln=value_[0][0][0]
                    if (value_[1]) {
                        value_[1].forEach((value__)=>{
                            if (value__[0]) value__[1]()
                            else if (Object.keys(event).includes(ln)) event[ln].push(value__[1])
                            else event[ln]=[value__[1]]
                        })
                    }
                    if (!index) {
                        press(lookup([ln,value_[0][0][1]-1]),"btn","－",config.color)
                        press(lookup([ln,value_[0][0][1]+list.length]),"btn","－",config.color)
                    }
                    for (i=0;i<list.length;i++) {
                        word=press(lookup([value_[0][0][0],value_[0][0][1]+i]),"menu",list[i],value_[0][2]||config.color)
                        if (!i) word.attr("menu",ln)
                    }
                })
            }
        })
    }
    $(document).ready(()=>{
        // home()
        goto("lounge",[2,3])
        $("game").parent().css("ln-height","1")
        $("z").click(function () {
            key(parseInt($(this).attr("class")))
        })
    }).keydown((e)=>{
        if ($.inArray(key(e.which),[37,38,39,40])!==-1) {
            e.preventDefault()
        }
    })
</script>
